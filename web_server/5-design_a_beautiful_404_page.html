#!/usr/bin/env bash # Configure your Nginx server to have a custom 404 page that contains the string `Ceci n'est pas une page` sudo apt-get update sudo apt-get install -y nginx echo " 404 - Ceci n'est pas une page

404

Ceci n'est pas une page


Opps! The page not found.

Find the hidden treasure!

" | sudo tee /var/www/html/my404.html new_error_page="listen 80 default_server;\n\terror_page 404 \/my404.html;\n\tlocation = \/my404.html {\n\t\troot \/var\/www\/html;\n\t\tinternal;\n\t}" sudo sed -i "s|listen 80 default_server;|$new_error_page|" /etc/nginx/sites-enabled/default sudo service nginx restart
